#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
   int t;
   cin>>t;
   while(t){
       int n;
       cin>>n;
       int a=n/3;
       int b=n/3;
       int c=n/3;
       int carry=n-(a+b+c);
       if(carry==1){
           a++;
       }
       else if(carry==2){
           a++;
           b++;
       }
       a++;
       c--;
       cout<<b<<" "<<a<<" "<<c<<endl;
       t--;
   }
}

#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t){ 
        vector<int>s;
        vector<int>e;
        int n;
        cin>>n;
        int i=0;
        while(i<n){
            int a;
            cin>>a;
            s.push_back(a);
            i++;
        }
        int j=0;
        int ma=0;
        while(j<n){
            int b;
            cin>>b;
            s[j]-=b;
            ma=max(ma,s[j]);
            e.push_back(b);
            j++;
        }
       int w=0;
       while(w<n){
           if(s[w]!=ma && e[w]){
               cout<<"NO"<<endl;
               break;
           }
           w++;
       }
       if(w==n){
           cout<<"YES"<<endl;
       }
        t--;
}
}

#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
   
    int t;
    cin>>t;
    while(t){ 
        vector<int>start;
    vector<int>end;
    vector<int>dur;
        int n;
        cin>>n;
        int i=0;
        while(i<n){
            int a;
            cin>>a;
            start.push_back(a);
            i++;
        }
        int j=0;
        while(j<n){
            int b;
            cin>>b;
            end.push_back(b);
            j++;
        }
        dur.push_back(end[0]-start[0]);
        int k=1;
        while(k<n){
            if(start[k]>=end[k-1]){
                dur.push_back(end[k]-start[k]);
            }
            else if(start[k]<end[k-1]){
                dur.push_back(end[k]-end[k-1]);
            }
            k++;
        }
        int h=0;
        while(h<n){
            cout<<dur[h]<<" ";
            h++;
        }
        cout<<endl;
        t--;
}
}

#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
   int t;
   cin>>t;
   while(t){
       int n;
       cin>>n;
       int k;
       cin>>k;
       string b;
       int i=0;
       while(i<n){
           char a;
           cin>>a;
           b.push_back(a);
           i++;
       }
       int sum=0;
       int ma=0;
       int j=0;
       while(j<k){
           if(b[j]=='B'){
               sum++;
           }
           j++;
       }
       ma=sum;
       int s=1;
       while(s+k-1<n){
           if(b[s-1]=='B'){
               sum--;
           }
           if(b[s+k-1]=='B'){
               sum++;
           }
            ma=max(ma,sum);
           s++;
       }
       cout<<k-ma<<endl;
       t--;
   }
}
